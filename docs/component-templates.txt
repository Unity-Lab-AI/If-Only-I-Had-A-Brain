    if (/^# /.test(line)) return '<h1>' + h.slice(2) + '</h1>';
    if (line.trim() === '') return '<br>';
    return '<div>' + h + '</div>';
  });
  preview.innerHTML = lines.join('');
};
input.addEventListener('input', render);
render();
